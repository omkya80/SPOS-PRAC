 
package ALGO;
import java.util.*;

class Process {
    int pid;      // process ID
    int burst;    // burst time
    int waiting;  // waiting time
    int tat;      // turnaround time

    Process(int pid, int burst) {
        this.pid = pid;
        this.burst = burst;
    }
}

public class SJF{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of processes: ");
        int n = sc.nextInt();

        Process[] processes = new Process[n];

        // Input burst times
        for (int i = 0; i < n; i++) {
            System.out.print("Enter Burst time for Process[" + (i + 1) + "]: ");
            int bt = sc.nextInt();
            processes[i] = new Process(i + 1, bt);
        }

        // Sort processes by burst time
        Arrays.sort(processes, Comparator.comparingInt(p -> p.burst));

        // Calculate WT & TAT
        int totalWT = 0, totalTAT = 0;
        processes[0].waiting = 0; // first process has no wait
        processes[0].tat = processes[0].burst;

        for (int i = 1; i < n; i++) {
            processes[i].waiting = processes[i - 1].waiting + processes[i - 1].burst;
            processes[i].tat = processes[i].waiting + processes[i].burst;

            totalWT += processes[i].waiting;
            totalTAT += processes[i].tat;
        }

        totalWT += processes[0].waiting;
        totalTAT += processes[0].tat;

        // Print results
        System.out.println("\nProcess\tBurst Time\tWaiting Time\tTurnaround Time");
        for (Process p : processes) {
            System.out.println("P" + p.pid + "\t\t" + p.burst + "\t\t" + p.waiting + "\t\t" + p.tat);
        }

        System.out.printf("\nAverage Waiting Time: %.2f", (double) totalWT / n);
        System.out.printf("\nAverage Turnaround Time: %.2f\n", (double) totalTAT / n);
    }
}
